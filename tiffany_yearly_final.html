renaming

drought_levels <- function(x) {
         (
          if (x==-1) {
            "No Drought (NONE)"
          } else if (x==0) {
            "Abnormally Dry (D0)"
          } else if (x==1) {
            "Moderate Drought (D1)"
          } else if (x==2) {
            "Severe Drought (D2)"
          } else if (x==3) {
            "Extreme Drought (D3)"
          } else
            "Exceptional Drought (D4)"
         )
}

bins <- bins %>% mutate(yearly_avg=sapply(yearly_avg,drought_levels))
